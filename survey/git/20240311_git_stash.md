# git stash

```sh
# 変更を退避する（`git stash push`と同等）
git stash
# ファイル単位で変更を退避する
git stash push -- <path>
# untracked ファイルも含める
git stash [-u|--include-untracked]
# untracked ファイルと gitignore ファイルも含める
git stash -a
# ステージしていないものだけ stash
git stash --keep-index
# stash 対象を対話的に選択
git stash --patch
# 変更を退避してコメントを残す
git stash save [-<u|a>] <comment>
# 退避の差分を見る
git stash show [n]
# 退避の差分を詳細に見る
git stash show -p [n]
# 退避を現在のブランチに反映する
git stash apply [n]
# ステージ状態も復元
git stash apply --index
# 退避を現在のブランチに反映してリストから削除する
git stash pop [n]
# 退避を削除する
git stash drop [n]
# すべての退避を削除する
git stash clear
```
